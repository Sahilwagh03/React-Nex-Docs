import React, { useState } from 'react';
import Button from '../components-react-nex/Button/Button'
import { Modal, ModalContent, ModalDescription, ModalFooter, ModalHeader, ModalTitle } from '../components-react-nex/Modal/Modal';

const ModalExample = () => {

const [modal, toggleModal] = useState(false)

  return (

    <Button onClick={() => toggleModal(true)} />
    {
        modal &&
        <Modal Open={modal}>
          <ModalContent>
            <ModalHeader>
              <ModalTitle>Are you absolutely sure?</ModalTitle>
              <ModalDescription> This action cannot be undone. This will permanently delete your
            account and remove your data from our servers.</ModalDescription>
            </ModalHeader>
            <ModalFooter>
              <Button title='Close' onClick={() => toggleModal(false)} className='bg-transparent border-2 dark:border-[#09090b]'></Button>
              <Button title='Continue' onClick={() => toggleModal(false)}></Button>
            </ModalFooter>
          </ModalContent>
        </Modal>
    }
  );
};

export default ModalExample;

`}
